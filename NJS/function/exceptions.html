Function Exceptions 函数异常

形式及定义

JS 提供了一套异常处理机制。异常是干扰程序的正常流程的不寻常的事故。
throw 语句中断函数的执行。它应该抛出一个 exception 对象。该对象包含一个用来识别异常类型的 name 属性和一个描述性的 message 属性。你也可以添加其他的属性。 

                        
    // 当参数不是数字类型时, 抛出异常

    var add = function (a) {
        if (typeof a !== 'number') {
            throw {
                name: 'TypeError',
                message: 'add needs numbers'
            }
        }
    };


    // 处理抛出的异常

    var try_it = function () {
        try {
            add('ss');
        } catch (e) {

        }
    };

    try_it();

用途

处理转换 JSON 

                        
    // 转换 json
    
    var json = function () {
        var obj;

        try {
            JSON.parse(obj);
        } catch (e) {
            obj = {};
        }

        return obj;
    }
